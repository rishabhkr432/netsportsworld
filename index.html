  •  function extractDomain($text){
    if ($text === null){
        return "No text entered";
    }
    else{
        $url = parse_url($text);   // parse url
        $domain = $url['host'];     // extract host
        $domainName = str_replace('www.','',$domain); // remove wwww. and return string
        return $domainName;
    }
}
$text = "https://www.google.com";

echo extractDomain($text);
